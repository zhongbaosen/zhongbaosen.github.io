## 安装步骤

### 安装准备

由于我的vue项目部署需要容器 同时也需要反向代理服务器接口 就写了这个教程进行分享

### 安装Nginx

部署之前，最好检查一下是否已经安装有nginx

这里我用的是sh文件一键安装的，就不阐述过程了

以下是Nginx配置

```bash
server
        {
                listen 80 default;
                #listen [::]:80 default ipv6only=on;
                server_name platform.senshine.top;
                index index.html index.htm index.php;
                root  /home/wwwroot/default;

                #error_page   404   /404.html;

    location ^~/haha/ { #添加访问目录为/proxy的代理配置
        rewrite  ^.+/haha/?(.*)$   /$1  break;
        proxy_pass [your url];
    }

    location /user/login { #添加访问目录为/bproxy的代理配置
        rewrite  ^.+/user/login/?(.*)$   /$1  break;
        proxy_pass [your url];
    }

    location /get_state { #添加访问目录为/aproxy的代理配置
        rewrite  ^.+/get_state/?(.*)$   /$1  break;
        proxy_pass [your url];
    }

    location /get_list { #添加访问目录为/aproxy的代理配置
        rewrite  ^.+/get_list/?(.*)$   /$1  break;
        proxy_pass [your url];
    }

                access_log  /home/wwwlogs/access.log  access;
        }
include vhost/*.conf;
}
```
